        private void DistributeCards(Random rnd)
        {
            List<Card> remaining = new List<Card>();

            for (int suspect = (int)Card.SuspectID.MinSuspectID; suspect <= (int)Card.SuspectID.MaxSuspectID; suspect++)
            {
                if(suspect == hiddenCards[0].Id || suspect == hiddenCards[1].Id || suspect == hiddenCards[2].Id)
                {
                    continue;
                }
                remaining.Add(new Card(suspect));

            }
            for (int weapon = (int)Card.WeaponID.MinWeaponID; weapon <= (int)Card.WeaponID.MaxWeaponID; weapon++)
            {
                if (weapon == hiddenCards[0].Id || weapon == hiddenCards[1].Id || weapon == hiddenCards[2].Id)
                {
                    continue;
                }
                remaining.Add(new Card(weapon));
            }
            for (int room = (int)Card.RoomID.MinRoomID; room <= (int)Card.RoomID.MaxRoomID; room++)
            {
                if (room == hiddenCards[0].Id || room == hiddenCards[1].Id || room == hiddenCards[2].Id)
                {
                    continue;
                }
                remaining.Add(new Card(room));
            }
            //
            PlayerCards = new List<List<Card>>();
            int player;
            for (player = 0; player < Players.NumPlayers; player++)
            {
                PlayerCards.Add(new List<Card>());
            }

            player = 0;
            while (remaining.Count > 0)
            {
                int idx = rnd.Next(remaining.Count);
                Card card = remaining[idx];
                PlayerCards[player].Add(card);
                player = (player + 1) % Players.NumPlayers;
                //System.Diagnostics.Debug.WriteLine(card);
                remaining.RemoveAt(idx);
            }

        }

        public bool Accuse(Card[] trio)
        {
            if (trio[0].Id != hiddenCards[0].Id && trio[0].Id != hiddenCards[1].Id && trio[0].Id != hiddenCards[2].Id)
            {
                return false;
            }
            if (trio[1].Id != hiddenCards[0].Id && trio[1].Id != hiddenCards[1].Id && trio[1].Id != hiddenCards[2].Id)
            {
                return false;
            }
            if (trio[2].Id != hiddenCards[0].Id && trio[2].Id != hiddenCards[1].Id && trio[2].Id != hiddenCards[2].Id)
            {
                return false;
            }
            return true;
        }


        public Card MatchOneCardOfPlayer(List<Card> list, Card[] trio)
        {
            foreach(Card card in list)
            {
                if(card.Id == trio[0].Id || card.Id == trio[1].Id || card.Id == trio[2].Id)
                {
                    return card;
                }
            }
            return null;
        }
        public Card Suggest(int player, Card[] trio)
        {
            return MatchOneCardOfPlayer(PlayerCards[player], trio);
        }
